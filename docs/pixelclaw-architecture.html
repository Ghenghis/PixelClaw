🔴 WHAT'S COMPLETELY MISSING
PixelClaw Gateway (FastAPI)
Nothing connects PixelAgents UI to ZeroClaw. No bridge service exists. This is the #1 blocker.
Provider Abstraction

🖥️
LM Studio
Type: Local / OpenAI-compatible
Endpoint: localhost:1234/v1
Auth: api_key: lm-studio (dummy)
Models (3090 Ti 24GB):
· Llama 3.3 70B Q4 (~20GB)
· Qwen 2.5 72B Q4 (~20GB)
· Mistral 7B Q8 (~8GB, fastest)
· nomic-embed-text (embeddings)
Cost: Free · 100% private · Offline
Best for: All tasks by default

Claude API
Type: Cloud / Anthropic API
Endpoint: api.anthropic.com
Auth: ANTHROPIC_API_KEY in .env
Models:
· claude-sonnet-4-6 (balanced)
· claude-opus-4-6 (complex reasoning)
· claude-haiku-4-5 (fast/cheap)
Cost: Per-token pricing
Best for: Complex multi-step reasoning, long context
💻
Claude Code
Type: VS Code Extension
Endpoint: Extension API / CLI subprocess
Auth: Handled by Claude Code extension
Integration:
· Trigger via VS Code commands API
· Read Claude Code diffs + suggestions
· PixelAgent character reacts to edits
Best for: Code editing, refactoring, file ops within VS Code workspace

Provider Router Python Class (Gateway Core)
# gateway/providers/router.py from openai import AsyncOpenAI import anthropic, yaml, os class ProviderRouter: def __init__(self, config_path: str): with open(config_path) as f: self.cfg = yaml.safe_load(f) self.provider = self.cfg["provider"]["default"] def get_client(self) -> AsyncOpenAI | anthropic.AsyncAnthropic: if self.provider == "lmstudio": cfg = self.cfg["provider"]["lmstudio"] return AsyncOpenAI( base_url=cfg["base_url"], api_key="lm-studio" ) elif self.provider == "claude": key_env = self.cfg["provider"]["claude"]["api_key_env"] return anthropic.AsyncAnthropic(api_key=os.environ[key_env]) raise ValueError(f"Unknown provider: {self.provider}") def get_model(self) -> str: return self.cfg["provider"][self.provider]["model"]

🔗 GATEWAY — Core FastAPI App
# gateway/main.py from fastapi import FastAPI, WebSocket from fastapi.middleware.cors import CORSMiddleware from .providers.router import ProviderRouter from .tools.zeroclaw_tool import ZeroClawTool from .memory.store import MemoryStore from .jsonl_emitter import JSONLEmitter import asyncio, json, yaml app = FastAPI(title="PixelClaw Gateway") app.add_middleware(CORSMiddleware, allow_origins=["*"]) router = ProviderRouter(".pixelclaw.yaml") zclaw = ZeroClawTool() memory = MemoryStore("data/memory.db") emitter = JSONLEmitter("pixelagents/events/") @app.get("/health") async def health(): return {"status":"ok","provider":router.provider} @app.websocket("/ws/chat") async def chat_ws(ws: WebSocket): await ws.accept() while True: msg = await ws.receive_json() ctx = await memory.retrieve(msg["text"]) await emitter.emit("thinking", msg["text"]) async for chunk in router.stream(msg, ctx): await ws.send_json({"delta": chunk}) await memory.store(msg["text"]) await emitter.emit("idle", "")
⚙ ZEROCLAW TOOL WRAPPER — Subprocess with JSONL
# gateway/tools/zeroclaw_tool.py import asyncio, json, subprocess from pathlib import Path class ZeroClawTool: def __init__(self, binary="zeroclaw"): self.binary = binary async def run( self, task: str, project_dir: str, timeout: float = 120.0 ) -> dict: proc = await asyncio.create_subprocess_exec( self.binary, "agent", "-m", task, "--json-output", # structured stdout cwd=project_dir, stdout=asyncio.subprocess.PIPE, stderr=asyncio.subprocess.PIPE ) try: out, err = await asyncio.wait_for( proc.communicate(), timeout=timeout ) except asyncio.TimeoutError: proc.kill() return {"error":"timeout","task":task} try: return json.loads(out) except json.JSONDecodeError: # fallback: wrap raw stdout return {"result": out.decode(),"raw":True}
🎬 JSONL EMITTER — Trigger PixelAgents Animations
# gateway/jsonl_emitter.py # PixelAgents watches a folder for .jsonl events # Each line = one agent activity event import json, aiofiles from datetime import datetime, timezone from pathlib import Path # PixelAgents event types: # thinking | working | typing | reading | idle | done | error ANIM_MAP = { "thinking": "walk_to_desk", "working": "typing", "reading": "look_at_screen", "done": "stretch", "idle": "idle_loop", "error": "confused", } class JSONLEmitter: def __init__(self, watch_dir: str): self.path = Path(watch_dir) / "events.jsonl" self.path.parent.mkdir(parents=True, exist_ok=True) async def emit(self, event_type: str, text: str): event = { "ts": datetime.now(timezone.utc).isoformat(), "type": event_type, "anim": ANIM_MAP.get(event_type,"idle"), "text": text[:200], # truncate for display } async with aiofiles.open(self.path,"a") as f: await f.write(json.dumps(event) +"\n")
